148void WorkerPool::stop()
149{
150 if (!running)
151 return;
152
153 running = false;
154
155 Logger::info("[Worker Pool] Stop requested, waiting for remaining tasks...");
156 auto start = std::chrono::steady_clock::now();
157 while (!task_queue.empty())
158 {
159 std::this_thread::sleep_for(std::chrono::milliseconds(5));
160
161 if (std::chrono::steady_clock::now() - start > std::chrono::seconds(1))
162 {
163 Logger::warn("[Worker Pool] Timeout waiting for queue to drain.");
164 break;
165 }
166 }
167
168 task_queue.close();
169 Logger::info("[Worker Pool] Task queue drained, closing...");
170
171 for (auto& worker_pair : workers)
172 {
173 auto& thread = worker_pair.second;
174 if (thread.joinable())
175 thread.join();
176 }
177
178 workers.clear();
179}
