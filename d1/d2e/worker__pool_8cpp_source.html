72void WorkerPool::stop() {
73 if (!running) return;

75 running = false;
76
77 Logger::info("[Worker Pool] Stop requested, waiting for remaining tasks...");
78 auto start = std::chrono::steady_clock::now();
79 while (!task_queue.empty()) {
80 std::this_thread::sleep_for(std::chrono::milliseconds(5));
81
82 if (std::chrono::steady_clock::now() - start > std::chrono::seconds(1)) {
83 Logger::warn("[Worker Pool] Timeout waiting for queue to drain.");
84 break;
85 }
86 }
87
88 task_queue.close();
89 Logger::info("[Worker Pool] Task queue drained, closing...");
90
91 for (auto& worker_pair : workers) {
92 auto& thread = worker_pair.second;

94 }
