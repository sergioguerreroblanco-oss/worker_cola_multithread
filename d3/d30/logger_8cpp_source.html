62std::string Logger::timestamp() {
63 using clock = std::chrono::system_clock;
64 const auto now = clock::now();
65 const std::time_t tt = clock::to_time_t(now);
66
67 std::tm tm{};
68#if defined(_WIN32)
69 localtime_s(&tm, &tt);
70#else
71 localtime_r(&tt, &tm);
72#endif
73
74 std::ostringstream oss;
75 oss << std::put_time(&tm, "%Y-%m-%d %H:%M:%S");
