165std::string Logger::timestamp()
166{
167 using clock = std::chrono::system_clock;
168 const auto now = clock::now();
169 const std::time_t tt = clock::to_time_t(now);
170
171 std::tm tm{};
172#if defined(_WIN32)
173 localtime_s(&tm, &tt);
174#else
175 localtime_r(&tt, &tm);
176#endif
177
178 std::ostringstream oss;
179 oss << std::put_time(&tm, "%Y-%m-%d %H:%M:%S");
180 return oss.str();
